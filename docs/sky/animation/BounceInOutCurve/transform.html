Return the value of the curve at point t

The value of t must be between 0.0 and 1.0, inclusive.

double transform(double t) {
  if (t < 0.5)
    return (1.0 - _bounce(1.0 - t)) * 0.5;
  else
    return _bounce(t * 2.0 - 1.0) * 0.5 + 0.5;
}
