void acceptGesture(int pointer) {
  if (_state != DragState.accepted) {
    _state = DragState.accepted;
    T delta = _pendingDragDelta;
    _pendingDragDelta = _initialPendingDragDelta;
    if (onStart != null)
      onStart();
    if (delta != _initialPendingDragDelta && onUpdate != null)
      onUpdate(delta);
  }
}
