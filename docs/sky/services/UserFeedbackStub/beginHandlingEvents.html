void beginHandlingEvents() {
  if (!isBound) {
    throw new MojoApiError("MojoEventHandler is unbound.");
  }
  _isOpen = true;
  _eventSubscription.subscribe((List<int> event) {
    try {
      _handleEvent(event);
    } catch (e) {
      close(immediate: true).then((_) {
        if (onError != null) {
          onError(e);
        }
      });
    }
  });
}
