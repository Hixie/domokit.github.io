String toString() {
  switch(this) {
    case STANDARD:
      return 'SystemUiVisibility.STANDARD';
    case FULLSCREEN:
      return 'SystemUiVisibility.FULLSCREEN';
    case IMMERSIVE:
      return 'SystemUiVisibility.IMMERSIVE';
  }
}
