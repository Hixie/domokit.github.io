static CompletionData deserialize(bindings.Message message) {
  var decoder = new bindings.Decoder(message);
  var result = decode(decoder);
  if (decoder.excessHandles != null) {
    decoder.excessHandles.forEach((h) => h.close());
  }
  return result;
}
