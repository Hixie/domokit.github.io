double transform(double t, Direction direction) {
  Interval interval = _getInterval(direction);
  if (interval != null)
    t = interval.transform(t);
  if (t == 1.0) // Or should we support inverse curves?
    return t;
  Curve curve = _getCurve(direction);
  if (curve != null)
    t = curve.transform(t);
  return t;
}
