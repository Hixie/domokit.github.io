bool retainStatefulNodeIfPossible(StatefulComponent newNode) {
  assert(!_disqualifiedFromEverAppearingAgain);
  assert(newNode != null);
  assert(_canSync(this, newNode));
  assert(_child != null);
  newNode._disqualifiedFromEverAppearingAgain = true;

  newNode._child = _child;
  _child = null;
  _dirty = true;

  return true;
}
