void fling(Offset velocity) {
  if (velocity != Offset.zero) {
    _startToEndAnimation(velocity: _scrollVelocity(velocity));
  } else if (!_toEndAnimation.isAnimating && (_toOffsetAnimation == null || !_toOffsetAnimation.isAnimating)) {
    settleScrollOffset();
  }
}
