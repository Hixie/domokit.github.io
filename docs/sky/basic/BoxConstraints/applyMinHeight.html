BoxConstraints applyMinHeight(double newMinHeight) {
  return new BoxConstraints(minWidth: minWidth,
                            maxWidth: maxWidth,
                            minHeight: math.max(minHeight, newMinHeight),
                            maxHeight: math.max(maxHeight, newMinHeight));
}
