static Route of(BuildContext context) {
  Route result;
  context.visitAncestorElements((Element element) {
    if (element is StatefulComponentElement && element.state is _RouteWidgetState) {
      result = element.widget.route;
      return false;
    }
    return true;
  });
  return result;
}
