String toString() {
  String result = 'Paint(color:$color';
  if (shader != null)
    result += ', shader: $shader';
  if (colorFilter != null)
    result += ', colorFilter: $colorFilter';
  if (maskFilter != null)
    result += ', maskFilter: $maskFilter';
  // TODO(mpcomplete): Figure out how to show a drawLooper.
  if (drawLooper != null)
    result += ', drawLooper:true';
  result += ')';
  return result;
}
